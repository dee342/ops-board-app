'use strict'; angular.module('OpsBoard') .controller('ActionPanelCtrl', function($scope, OpsBoardRepository, $rootScope) { $scope.equipments = OpsBoardRepository.getEquipment(); $scope.persons = OpsBoardRepository.getPersonnel(); $scope.getActions = function () { return OpsBoardRepository.getActions(); } $scope.getOnlineUsers = function () { return OpsBoardRepository.getOnlineUsers(); } $scope.actionDay = ""; $scope.createHeader = function(index, day) { if(index == 0){ $scope.actionDay = ""; } var showHeader = false; if(day !== $scope.actionDay){ showHeader = true; } $scope.actionDay = day; return showHeader; } });
